 -- Users
CREATE TABLE users(
user_id SERIAL PRIMARY KEY NOT NULL,
firstname character varying (50) NOT NULL,
lastname character varying (255) NOT NULL,
password character varying (255),
mobile_number BIGINT,
email character varying(100),
active boolean DEFAULT true NOT NULL,
isDeleted boolean DEFAULT false NOT NULL,
createddate timestamp without time zone DEFAULT ('now'::text)::date,
updatedDate timestamp without time zone,
username character varying (255);
);



-- Roles
create table roles(
role_id SERIAL PRIMARY KEY NOT NULL,
role_name character varying (15),
role_description character varying (55),
active boolean DEFAULT true NOT NULL,
isDeleted boolean DEFAULT false NOT NULL,
createddate timestamp without time zone DEFAULT ('now'::text)::date NOT NULL,
lastmodifieddate timestamp without time zone
)

-- Role Mapping
CREATE TABLE user_role_mapping(
id SERIAL PRIMARY KEY NOT NULL,
user_id Integer,
role_id Integer,
active boolean DEFAULT true NOT NULL,
createddate timestamp without time zone DEFAULT ('now'::text)::date NOT NULL,
lastmodifieddate timestamp without time zone,
FOREIGN KEY (user_id) REFERENCES users(user_id),
FOREIGN KEY (role_id) REFERENCES roles(role_id)
)



CREATE TABLE mst_products(
product_id SERIAL PRIMARY KEY NOT NULL,
product_name  character varying (255),
product_image  TEXT,
short_description character varying (255),
long_description TEXT,
product_tax character varying (2),
product_price character varying (2),
active boolean DEFAULT true NOT NULL,
is_deleted boolean DEFAULT false NOT NULL,
createddate timestamp without time zone DEFAULT ('now'::text)::date,
updated_date timestamp without time zone
);

 CREATE TABLE mst_products_thumbnails (
     thumbnail_id serial PRIMARY KEY  NOT NULL,
     product_id INTEGER NOT NULL,
     document_name character varying(255) NOT NULL,
     uploaded_by INTEGER NOT NULL,
     active boolean DEFAULT true NOT NULL,
     createddate timestamp without time zone DEFAULT ('now'::text)::date NOT NULL,
     lastmodifieddate timestamp without time zone,
     FOREIGN KEY (product_id) REFERENCES mst_products(product_id)
 );

 -- Create the wishlist table
 CREATE TABLE wishlist (
     wishlist_id SERIAL PRIMARY KEY,
     user_id INT,
     product_id INT,
     quantity INT DEFAULT 1,
 	createddate timestamp without time zone DEFAULT ('now'::text)::date,
     active boolean DEFAULT true NOT NULL,
 	FOREIGN KEY (user_id) REFERENCES users(user_id),
     FOREIGN KEY (product_id) REFERENCES mst_products(product_id)
 );


  -- Create the Address
  CREATE TABLE mst_address (
      address_id SERIAL PRIMARY KEY,
      user_id INT,
      country_id INT,
      state_id INT,
      city  character varying(255) NOT NULL,
      pincode  character varying(255) NOT NULL,
      address  character varying(255) NOT NULL,
      default_address  boolean DEFAULT true NOT NULL,
  	  createddate timestamp without time zone DEFAULT ('now'::text)::date,
      active boolean DEFAULT true NOT NULL,
  	  FOREIGN KEY (state_id) REFERENCES mst_state(state_id),
  	   FOREIGN KEY (user_id) REFERENCES users(user_id)
  );


--Carts
  CREATE TABLE mst_cart (
      cart_id SERIAL PRIMARY KEY,
      user_id INTEGER,
	  quantity_id INTEGER,
	  status  character varying(255) NOT NULL,
  	  createddate timestamp without time zone DEFAULT ('now'::text)::date,
      active boolean DEFAULT true NOT NULL,
  	  FOREIGN KEY (user_id) REFERENCES users(user_id)
  );


-- cart items
--Carts
  CREATE TABLE mst_cartitem (
      cartitem_id SERIAL PRIMARY KEY,
      user_id INTEGER,
      cart_id INTEGER,
	  product_id INTEGER,
	  product_quantity INTEGER,
  	  createddate timestamp without time zone DEFAULT ('now'::text)::date,
      active boolean DEFAULT true NOT NULL,
      FOREIGN KEY (cart_id) REFERENCES mst_cart(cart_id),
  	  FOREIGN KEY (product_id) REFERENCES mst_products(product_id),
  	  FOREIGN KEY (user_id) REFERENCES users(user_id)
  );


  --Orders

  CREATE TABLE mst_orders (
         id SERIAL PRIMARY KEY,
         amount double precision,
         currency character varying(50) NOT NULL,
   	    receipt character varying(100) NOT NULL,
   	    order_id character varying(255) NOT NULL,
   	    alternate_phone character varying(20) NOT NULL,
   	    phone character varying(20) NOT NULL,
   	    additional_info character varying(255) NOT NULL,
   	    payment_methord character varying(100) NOT NULL,
   	    shippingaddress_id INTEGER,
   	    billingaddress_id INTEGER,
   	    user_id INTEGER,
   	    cart_id INTEGER,
   	    gateway_entity character varying(100) NOT NULL,
   	    status character varying(50) NOT NULL,
   	    attempts INTEGER,
         createddate timestamp without time zone DEFAULT ('now'::text)::date,
         active boolean DEFAULT true NOT NULL,
         FOREIGN KEY (cart_id) REFERENCES mst_cart(cart_id),
     	FOREIGN KEY (user_id) REFERENCES users(user_id)
     );

       CREATE TABLE mst_queries (
              query_id SERIAL PRIMARY KEY,
              name character varying(255) NOT NULL,
              email character varying(255) NOT NULL,
              phone character varying(50) NOT NULL,
        	  support_for character varying(255) NOT NULL,
        	  message text,
              createddate timestamp without time zone DEFAULT ('now'::text)::date,
              active boolean DEFAULT true NOT NULL
          );
